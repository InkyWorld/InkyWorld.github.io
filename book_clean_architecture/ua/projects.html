from application.ports import PaymentGateway
from .stripe_gateway import StripeGateway
from .paypal_gateway import PayPalGateway
from .crypto_gateway import CryptoGateway

class PaymentGatewayFactory:
    """Factory –¥–ª—è —Å—Ç–≤–æ—Ä–µ–Ω–Ω—è –ø–ª–∞—Ç—ñ–∂–Ω–∏—Ö —à–ª—é–∑—ñ–≤"""

        self._config = config
        self._gateways = {}
    
    def get_gateway(self, method: PaymentMethod) -> PaymentGateway:
        if method not in self._gateways:
            self._gateways[method] = self._create_gateway(method)
        return self._gateways[method]
    
    def _create_gateway(self, method: PaymentMethod) -> PaymentGateway:
        if method == PaymentMethod.STRIPE:
            return StripeGateway(self._config["stripe_key"])
        elif method == PaymentMethod.PAYPAL:
            return PayPalGateway(
                self._config["paypal_client_id"],
                self._config["paypal_secret"]
            )
        elif method == PaymentMethod.CRYPTO:
            return CryptoGateway(self._config["crypto_wallet"])
        raise ValueError(f"Unknown payment method: {method}")

class EventBus:
    """Observer pattern implementation"""
    
    def __init__(self):
        self._handlers: dict[type, List[Callable]] = {}
    
    def subscribe(self, event_type: type, handler: Callable):
        """–ü—ñ–¥–ø–∏—Å–∫–∞ –Ω–∞ –ø–æ–¥—ñ—é"""
        if event_type not in self._handlers:
            self._handlers[event_type] = []
        self._handlers[event_type].append(handler)
    
    def publish(self, event: DomainEvent):
        """–ü—É–±–ª—ñ–∫–∞—Ü—ñ—è –ø–æ–¥—ñ—ó –≤—Å—ñ–º –ø–µ—Ä–µ–¥–ø–ª–∞—Ç–Ω–∏–∫–∞–º"""
        event_type = type(event)
        if event_type in self._handlers:
            for handler in self._handlers[event_type]:
                handler(event)

import logging
import time
from functools import wraps

class LoggingDecorator(NotificationSender):
    """Decorator –¥–ª—è –ª–æ–≥—É–≤–∞–Ω–Ω—è"""
    
    def __init__(self, sender: NotificationSender):
        self._sender = sender
        self._logger = logging.getLogger(__name__)

        self._logger.info(f"Sending to {recipient}")
        result = self._sender.send(recipient, message)
        self._logger.info(f"Result: {result}")
        return result

class RetryDecorator(NotificationSender):
    """Decorator –¥–ª—è –ø–æ–≤—Ç–æ—Ä–Ω–∏—Ö —Å–ø—Ä–æ–±"""
    
    def __init__(self, sender: NotificationSender, max_retries: int = 3):
        self._sender = sender
        self._max_retries = max_retries
    
    def send(self, recipient: str, message: str) -> bool:
        for attempt in range(self._max_retries):
            try:
                return self._sender.send(recipient, message)
            except Exception as e:
                if attempt == self._max_retries - 1:
                    raise
                time.sleep(2 ** attempt)  # Exponential backoff
        return False

# application/validators.py
from abc import ABC, abstractmethod
from domain.entities import UploadedFile
from domain.exceptions import ValidationError

class FileValidator(ABC):
    """Base handler in Chain of Responsibility"""
    
    def __init__(self):
        self._next: FileValidator = None
    
    def set_next(self, validator: 'FileValidator') -> 'FileValidator':
        self._next = validator
        return validator
    
    def validate(self, file: UploadedFile) -> bool:
        if self._check(file):
            if self._next:
                return self._next.validate(file)
            return True
        return False
    
    @abstractmethod
    def _check(self, file: UploadedFile) -> bool:
        pass

class FileSizeValidator(FileValidator):
    def __init__(self, max_size_mb: int = 10):
        super().__init__()
        self.max_size = max_size_mb * 1024 * 1024
    
    def _check(self, file: UploadedFile) -> bool:
        if file.size > self.max_size:
            raise ValidationError(f"File too large: {file.size} bytes")
        return True

class FileTypeValidator(FileValidator):
    def __init__(self, allowed_types: list):
        super().__init__()
        self.allowed_types = allowed_types
    
    def _check(self, file: UploadedFile) -> bool:
        if file.content_type not in self.allowed_types:
            raise ValidationError(f"Type not allowed: {file.content_type}")
        return True

class VirusScanValidator(FileValidator):
    def __init__(self, scanner):
        super().__init__()
        self.scanner = scanner
    
    def _check(self, file: UploadedFile) -> bool:
        if not self.scanner.is_clean(file.content):
            raise ValidationError("File contains malware")
        return True

# infrastructure/s3_storage.py ‚Äî Adapter –¥–ª—è AWS S3
import boto3

class S3StorageAdapter(FileStorage):
    """Adapter: –ø–µ—Ä–µ—Ç–≤–æ—Ä—é—î —ñ–Ω—Ç–µ—Ä—Ñ–µ–π—Å boto3 –≤ –Ω–∞—à FileStorage"""
    
    def __init__(self, bucket: str, region: str):
        self.bucket = bucket
        self.s3 = boto3.client('s3', region_name=region)
    
    def save(self, key: str, content: bytes) -> str:
        self.s3.put_object(Bucket=self.bucket, Key=key, Body=content)
        return f"https://{self.bucket}.s3.amazonaws.com/{key}"
    
    def get(self, key: str) -> bytes:
        response = self.s3.get_object(Bucket=self.bucket, Key=key)
        return response['Body'].read()
    
    def delete(self, key: str) -> bool:
        self.s3.delete_object(Bucket=self.bucket, Key=key)
        return True

from pathlib import Path

class LocalStorageAdapter(FileStorage):
    def __init__(self, base_path: str):
        self.base_path = Path(base_path)
        self.base_path.mkdir(parents=True, exist_ok=True)
    
    def save(self, key: str, content: bytes) -> str:
        path = self.base_path / key
        path.write_bytes(content)
        return str(path)
    
    def get(self, key: str) -> bytes:
        return (self.base_path / key).read_bytes()
    
    def delete(self, key: str) -> bool:
        (self.base_path / key).unlink()
        return True

from abc import ABC, abstractmethod

class OrderState(ABC):
    """State interface"""
    
    @abstractmethod
    def confirm(self, order: 'Order'): pass
    
    @abstractmethod
    def ship(self, order: 'Order'): pass
    
    @abstractmethod
    def deliver(self, order: 'Order'): pass
    
    @abstractmethod
    def cancel(self, order: 'Order'): pass

class PendingState(OrderState):
    def confirm(self, order):
        order.set_state(ConfirmedState())
        order.add_event(OrderConfirmed(order.id))
    
    def ship(self, order):
        raise InvalidOperation("Cannot ship pending order")
    
    def deliver(self, order):
        raise InvalidOperation("Cannot deliver pending order")
    
    def cancel(self, order):
        order.set_state(CancelledState())
        order.add_event(OrderCancelled(order.id))

class ConfirmedState(OrderState):
    def confirm(self, order):
        raise InvalidOperation("Already confirmed")
    
    def ship(self, order):
        order.set_state(ShippedState())
        order.add_event(OrderShipped(order.id))
    
    def deliver(self, order):
        raise InvalidOperation("Must ship first")
    
    def cancel(self, order):
        order.set_state(CancelledState())
        order.add_event(OrderCancelled(order.id, reason="cancelled_after_confirm"))

class ShippedState(OrderState):
    def confirm(self, order):
        raise InvalidOperation("Already shipped")
    
    def ship(self, order):
        raise InvalidOperation("Already shipped")
    
    def deliver(self, order):
        order.set_state(DeliveredState())
        order.add_event(OrderDelivered(order.id))
    
    def cancel(self, order):
        raise InvalidOperation("Cannot cancel shipped order")

class Command(ABC):
    @abstractmethod
    def execute(self): pass
    
    @abstractmethod
    def undo(self): pass

@dataclass
class AddItemCommand(Command):
    order: 'Order'
    item: 'OrderItem'
    
    def execute(self):
        self.order.add_item(self.item)
    
    def undo(self):
        self.order.remove_item(self.item)

@dataclass
class ChangeQuantityCommand(Command):
    order: 'Order'
    item_id: str
    new_quantity: int
    _old_quantity: int = None
    
    def execute(self):
        item = self.order.get_item(self.item_id)
        self._old_quantity = item.quantity
        item.quantity = self.new_quantity
    
    def undo(self):
        item = self.order.get_item(self.item_id)
        item.quantity = self._old_quantity

# application/command_handler.py
class OrderCommandHandler:
    def __init__(self):
        self._history: list[Command] = []
    
    def execute(self, command: Command):
        command.execute()
        self._history.append(command)
    
    def undo(self):
        if self._history:
            command = self._history.pop()
            command.undo()
    
    def undo_all(self):
        while self._history:
            self.undo()

# domain/discounts.py
from abc import ABC, abstractmethod
from decimal import Decimal
from typing import List

class Discount(ABC):
    """Component interface"""
    
    @abstractmethod
    def calculate(self, order: 'Order') -> Decimal:
        pass

class PercentageDiscount(Discount):
    """Leaf: –ø—Ä–æ—Ü–µ–Ω—Ç–Ω–∞ –∑–Ω–∏–∂–∫–∞"""
    
    def __init__(self, percent: Decimal):
        self.percent = percent
    
    def calculate(self, order: 'Order') -> Decimal:
        return order.subtotal * (self.percent / 100)

class FixedDiscount(Discount):
    """Leaf: —Ñ—ñ–∫—Å–æ–≤–∞–Ω–∞ –∑–Ω–∏–∂–∫–∞"""
    
    def __init__(self, amount: Decimal):
        self.amount = amount
    
    def calculate(self, order: 'Order') -> Decimal:
        return min(self.amount, order.subtotal)

class CompositeDiscount(Discount):
    """Composite: –∫–æ–º–±—ñ–Ω–∞—Ü—ñ—è –∑–Ω–∏–∂–æ–∫"""
    
    def __init__(self, mode: str = 'sum'):
        self.discounts: List[Discount] = []
        self.mode = mode  # 'sum', 'max', 'first'
    
    def add(self, discount: Discount):
        self.discounts.append(discount)
    
    def calculate(self, order: 'Order') -> Decimal:
        if not self.discounts:
            return Decimal(0)
        
        amounts = [d.calculate(order) for d in self.discounts]
        
        if self.mode == 'sum':
            return sum(amounts)
        elif self.mode == 'max':
            return max(amounts)
        elif self.mode == 'first':
            return amounts[0] if amounts else Decimal(0)

# domain/specifications.py
from abc import ABC, abstractmethod

class Specification(ABC):
    @abstractmethod
    def is_satisfied_by(self, candidate) -> bool:
        pass
    
    def __and__(self, other: 'Specification') -> 'Specification':
        return AndSpecification(self, other)
    
    def __or__(self, other: 'Specification') -> 'Specification':
        return OrSpecification(self, other)
    
    def __invert__(self) -> 'Specification':
        return NotSpecification(self)

class AndSpecification(Specification):
    def __init__(self, *specs):
        self.specs = specs
    
    def is_satisfied_by(self, candidate) -> bool:
        return all(s.is_satisfied_by(candidate) for s in self.specs)

class OrSpecification(Specification):
    def __init__(self, *specs):
        self.specs = specs
    
    def is_satisfied_by(self, candidate) -> bool:
        return any(s.is_satisfied_by(candidate) for s in self.specs)

# –ö–æ–Ω–∫—Ä–µ—Ç–Ω—ñ —Å–ø–µ—Ü–∏—Ñ—ñ–∫–∞—Ü—ñ—ó –¥–ª—è –∑–Ω–∏–∂–æ–∫
class MinOrderAmountSpec(Specification):
    def __init__(self, min_amount: Decimal):
        self.min_amount = min_amount
    
    def is_satisfied_by(self, order) -> bool:
        return order.subtotal >= self.min_amount

class CustomerLevelSpec(Specification):
    def __init__(self, required_level: str):
        self.required_level = required_level
    
    def is_satisfied_by(self, order) -> bool:
        return order.customer.level == self.required_level

class FirstOrderSpec(Specification):
    def is_satisfied_by(self, order) -> bool:
        return order.customer.orders_count == 0

class DiscountRuleBuilder:
    def __init__(self):
        self._discount = None
        self._specification = None
        self._priority = 0
        self._stackable = True
    
    def with_percentage(self, percent: Decimal) -> 'DiscountRuleBuilder':

        return self
    
    def with_fixed_amount(self, amount: Decimal) -> 'DiscountRuleBuilder':
        self._discount = FixedDiscount(amount)

    def when(self, spec: Specification) -> 'DiscountRuleBuilder':
        self._specification = spec
        return self

        self._priority = priority
        return self

        self._stackable = False

            discount=self._discount,
            specification=self._specification,
            priority=self._priority,
            stackable=self._stackable

# domain/sagas.py
class CreateOrderSaga:
    def __init__(self, orchestrator):
        self.orchestrator = orchestrator

    async def start(self, order_data):
        # –ö—Ä–æ–∫ 1: –°—Ç–≤–æ—Ä–∏—Ç–∏ –∑–∞–º–æ–≤–ª–µ–Ω–Ω—è
        order_id = await self.orchestrator.send_command("orders_service", "create_order", order_data)
        
        try:
            # –ö—Ä–æ–∫ 2: –ó–∞—Ä–µ–∑–µ—Ä–≤—É–≤–∞—Ç–∏ –æ–ø–ª–∞—Ç—É
            payment_id = await self.orchestrator.send_command("payments", "reserve_funds", {"order_id": order_id})
            
            # –ö—Ä–æ–∫ 3: –ó–∞—Ä–µ–∑–µ—Ä–≤—É–≤–∞—Ç–∏ —Ç–æ–≤–∞—Ä –Ω–∞ —Å–∫–ª–∞–¥—ñ
            await self.orchestrator.send_command("warehouse", "reserve_items", {"order_id": order_id})
            
        except Exception as e:
            # –ö–æ–º–ø–µ–Ω—Å–∞—Ü—ñ—è (Rollback) —É –∑–≤–æ—Ä–æ—Ç–Ω–æ–º—É –ø–æ—Ä—è–¥–∫—É
            await self.orchestrator.send_command("payments", "refund", {"order_id": order_id})
            await self.orchestrator.send_command("orders", "cancel", {"order_id": order_id})
            raise SagaFailed("Order creation failed, rolling back...")
