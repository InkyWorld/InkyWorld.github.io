import uuid
from domain.entities import Order
from domain.value_objects import Money, Email
from .interfaces import OrderRepository, EmailService

class CreateOrderUseCase:
    # DEPENDENCY INJECTION:
    # –ú–∏ –ø—Ä–æ—Å–∏–º–æ –±—É–¥—å-—è–∫—ñ —Ä–µ–∞–ª—ñ–∑–∞—Ü—ñ—ó —Ü–∏—Ö —ñ–Ω—Ç–µ—Ä—Ñ–µ–π—Å—ñ–≤.

        self, 
        repo: OrderRepository, 
        email_service: EmailService

        self.email_service = email_service

    def execute(self, command: CreateOrderCommand) -> OrderResultDTO:
        # 1. –ü—ñ–¥–≥–æ—Ç—É–≤–∞—Ç–∏ Value Objects
        # –Ø–∫—â–æ email –Ω–µ–∫–æ—Ä–µ–∫—Ç–Ω–∏–π, –ø–æ–º–∏–ª–∫–∞ –≤–∏–Ω–∏–∫–Ω–µ –¢–£–¢ (–∑–∞—Ö–∏—Å—Ç —ñ–Ω–≤–∞—Ä—ñ–∞–Ω—Ç—ñ–≤)

            order_id=str(order.id),
            total_amount=order.total_price.amount,
